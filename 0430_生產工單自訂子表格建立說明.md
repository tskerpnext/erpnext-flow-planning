# 生產工單自訂子表格建立說明

**日期**：2026-04-30  
**適用版本**：ERPNext v16 / Frappe v16  
**功能**：在生產工單「配置」頁面掛載生產排程子表格，並依成品物料組自動填入預設項目

---

## 一、功能概述

| 項目     | 說明                             |
| ------ | ------------------------------ |
| 子表格名稱  | Work Order Production Schedule |
| 掛載位置   | 生產工單 → 配置頁 → 時間區塊下方            |
| 自動填入觸發 | 選擇「成品」欄位時，依物料組自動帶入對應排程項目       |

### 物料組對應項目

| 物料組 | 自動填入項目 |
|--------|------------|
| 电缆产品 | 绝缘、打点、螺旋、对捻、1Y、2Y、3Y、4Y、编织、缠绕、外被、卷取 |
| 加工产品 | 裁线组、焊接组、端子组、组装组、成品检测、收线 |

---

## 二、實作步驟

### 步驟 1：建立自訂子表格 DocType

在非開發者模式下，DocType 必須設定 `custom=1`。

```python
import frappe
frappe.init(site='site1.local', sites_path='/home/erpnext/frappe-bench/sites')
frappe.connect()
frappe.set_user('Administrator')

DOCTYPE_NAME = 'Work Order Production Schedule'

doc = frappe.get_doc({
    'doctype': 'DocType',
    'name': DOCTYPE_NAME,
    'module': 'Manufacturing',
    'istable': 1,       # 宣告為子表格
    'custom': 1,        # 非開發者模式必須設為 1
    'editable_grid': 1,
    'fields': [
        {
            'fieldname': 'project_name',
            'label': '項目',
            'fieldtype': 'Data',
            'in_list_view': 1,
            'columns': 4,
            'length': 140,
        },
        {
            'fieldname': 'due_date',
            'label': '預計完工日',
            'fieldtype': 'Date',
            'in_list_view': 1,
            'columns': 4,
        },
        {
            'fieldname': 'finish_date',
            'label': '實際完工日',
            'fieldtype': 'Date',
            'in_list_view': 1,
            'columns': 4,
        },
    ],
    'permissions': [
        {'role': 'Manufacturing User', 'read': 1, 'write': 1, 'create': 1, 'delete': 1},
        {'role': 'System Manager',     'read': 1, 'write': 1, 'create': 1, 'delete': 1},
    ],
})
doc.insert(ignore_permissions=True)
frappe.db.commit()
frappe.destroy()
```

### 步驟 2：新增 Section Break（分隔線）

在「時間」區塊最後一個欄位 `lead_time` 後加入分隔線，讓子表格獨佔整行。

```python
import frappe
frappe.init(site='site1.local', sites_path='/home/erpnext/frappe-bench/sites')
frappe.connect()
frappe.set_user('Administrator')

# Section Break
sb = frappe.get_doc({
    'doctype': 'Custom Field',
    'dt': 'Work Order',
    'fieldname': 'custom_production_schedule_section',
    'label': '生產排程',
    'fieldtype': 'Section Break',
    'insert_after': 'lead_time',   # 時間區塊的最後一個欄位
})
sb.insert(ignore_permissions=True)

# 子表格欄位
cf = frappe.get_doc({
    'doctype': 'Custom Field',
    'dt': 'Work Order',
    'fieldname': 'custom_production_schedule',
    'label': '生產排程',
    'fieldtype': 'Table',
    'options': 'Work Order Production Schedule',
    'insert_after': 'custom_production_schedule_section',
})
cf.insert(ignore_permissions=True)

frappe.db.commit()
frappe.destroy()
```

> **注意**：`insert_after` 填入前一個欄位的 `fieldname`，決定顯示順序。  
> 查詢配置頁欄位順序可用：
> ```python
> frappe.get_all('DocField',
>     filters={'parent': 'Work Order', 'fieldtype': ['in', ['Tab Break', 'Section Break']]},
>     fields=['fieldname', 'label', 'fieldtype'],
>     order_by='idx asc'
> )
> ```

### 步驟 3：建立 Client Script（自動填入邏輯）

```python
import frappe
frappe.init(site='site1.local', sites_path='/home/erpnext/frappe-bench/sites')
frappe.connect()
frappe.set_user('Administrator')

SCRIPT_BODY = """
const SCHEDULE_MAP = {
  '电缆产品': [
    '绝缘','打点','螺旋','对捻',
    '1Y','2Y','3Y','4Y',
    '编织','缠绕','外被','卷取'
  ],
  '加工产品': [
    '裁线组','焊接组','端子组','组装组','成品检测','收线'
  ]
};

function fill_schedule(frm, item_group) {
  const rows = SCHEDULE_MAP[item_group];
  frm.clear_table('custom_production_schedule');
  if (rows) {
    rows.forEach(name => {
      let row = frm.add_child('custom_production_schedule');
      row.project_name = name;
    });
  }
  frm.refresh_field('custom_production_schedule');
}

function fetch_and_fill(frm) {
  const item = frm.doc.production_item;
  if (!item) return;

  frappe.after_ajax(() => {
    if (frm.doc.production_item !== item) return;
    frappe.db.get_value('Item', item, 'item_group')
      .then(r => {
        if (frm.doc.production_item !== item) return;
        const ig = r && r.message && r.message.item_group;
        if (ig && SCHEDULE_MAP[ig]) {
          fill_schedule(frm, ig);
        }
      });
  });
}

frappe.ui.form.on('Work Order', {
  production_item: function(frm) {
    fetch_and_fill(frm);
  },
  onload: function(frm) {
    if (frm.is_new()) return;
  }
});
"""

cs = frappe.get_doc({
    'doctype': 'Client Script',
    'name': 'Work Order - 生產排程自動填入',
    'dt': 'Work Order',
    'view': 'Form',
    'enabled': 1,
    'script': SCRIPT_BODY,
})
cs.insert(ignore_permissions=True)
frappe.db.commit()
frappe.destroy()
```

## 三、注意事項

### 非開發者模式建立 DocType
若未設定 `custom=1`，會出現以下錯誤：
```
CannotCreateStandardDottypeError: Not in Developer Mode!
```
解法：在 DocType 定義中加入 `'custom': 1`。

### Custom Field 與既有欄位同名
若 DocType 原本就有該 fieldname，無法再建立同名 Custom Field，需改用 **Property Setter**：
```python
ps = frappe.get_doc({
    'doctype': 'Property Setter',
    'doctype_or_field': 'DocField',
    'doc_type': 'Work Order',
    'field_name': 'use_multi_level_bom',
    'property': 'default',
    'property_type': 'Text',
    'value': '0',
})
ps.insert(ignore_permissions=True)
frappe.db.commit()
```

### 腳本執行環境
- 以上 Python 腳本須在 `frappe-bench` 的 Python 環境下執行：`./env/bin/python script.py`
- 或透過 `bench execute` 執行單行指令

### Client Script 競爭條件（Race Condition）
- 使用 `frappe.after_ajax()` 等待 ERPNext 內部 AJAX 完成後再填入
- 雙重確認 `frm.doc.production_item === item`，避免快速切換物料時填入錯誤資料

## 四、相關設定

### 採用多級物料清單預設關閉

若 BOM 內含有已是成品的子物料（如 SK-002-1），多階展開會導致工單所需物料異常。  
建議將「採用多級物料清單」預設改為不勾選：

```python
ps = frappe.get_doc({
    'doctype': 'Property Setter',
    'doctype_or_field': 'DocField',
    'doc_type': 'Work Order',
    'field_name': 'use_multi_level_bom',
    'property': 'default',
    'property_type': 'Text',
    'value': '0',
})
ps.insert(ignore_permissions=True)
frappe.db.commit()
```
